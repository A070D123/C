# 函數設計
### 計算費次序列
```
#include <stdio.h>
#include <stdlib.h>

int fib_for(int x)
{
   int y=1,z=1,w=0; 
   if (x==1||x==2)return 1;
   for(int i=3;i<=x;i++) 
     { 
      w=y+z; 
      z=y;
	  y=w; 
     } 
  return w;
}	
/////////////////////////////////////
int fib_while(int x)
{
 int i=3,y=1,z=1,w=0;
 if (x==1||x==2)return 1;
   while(i<=x)
	    {
	     w=y+z; 
         z=y;
	     y=w; 
		 i++; 
		}	
 return w;
}
////////////////////////////////
int fib_do_while(int x)
{
  int i=3,y=1,z=1,w=0;
  if (x==1||x==2)return 1;
    do{
       	w=y+z; 
        z=y;
	    y=w; 
        i++; 
      }while(i<=x);
 return w; 	
}
////////////////////////////////////
int fib_recursion(int x)
{
    if (x==1||x==2) return 1;
    return fib_recursion(x-1)+fib_recursion(x-2);
}
///////////////////////////////////
int main()
{	
	int x;
	printf("請輸入數字:");
	scanf("%d",&x);
	printf("fib_for      第%d項是%d \n",x,fib_for(x));
	printf("fib_while    第%d項是%d \n",x,fib_for(x));
	printf("fib_do_while 第%d項是%d \n",x,fib_do_while(x));
	printf("fib_recursion第%d項是%d \n",x,fib_recursion(x));
	system("pause");
	return 0;
}
```
