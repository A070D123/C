# 1+3+....+101 總和 FOR
```
int main(void)
{
 int i,sum=0;
 for(i=1;i<=101;i++)			/* 計算1+3+...+99的結果 */
    sum+=i++;
 printf("1+3+5+...+101=%d\n",sum);	/* 印出sum的值 */
  
 system("pause");
 return 0;
}
```

# 1+3+..+101 總和 While Loop
```
#include <stdio.h>
#include <stdlib.h>
int main(void)
{
   int i=1,sum=0;	/* 設定迴圈初值 */
   while(i<=101)    /* while迴圈，當sum小於100則繼續累加 */
   {
      sum+=i;
      printf("從1累加到%2d=%2d\n",i,sum);
      i+=2;
   }
   system("pause");
   return 0;
}
```
# 1+3+..+101 總和 DO WHILE LOOP
```
#include <stdio.h>
#include <stdlib.h>
int main(void)
{
   int i=1,sum=0;	
   do
{
     sum+=i;  
     i+=2;   
}  while(i<=101);   
   printf("%d\n",sum);
   system("pause");
   return 0;
}
```
