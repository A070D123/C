# 程式設計
### 1+3+..+101 總和 While Loop
```
#include <stdio.h>
#include <stdlib.h>
int main(void)
{
   int i=1,sum=0;	/* 設定迴圈初值 */
   while(i<=101)    /* while迴圈，當sum小於100則繼續累加 */
   {
      sum+=i;
      printf("從1累加到%2d=%2d\n",i,sum);
      i+=2;
   }
   system("pause");
   return 0;
}
```
### 1+3+..+101 總和 DO WHILE LOOP
```
#include <stdio.h>
#include <stdlib.h>
int main(void)
{
   int i=1,sum=0;	
   do
{
     sum+=i;  
     i+=2;   
}  while(i<=101);   
   printf("%d\n",sum);
   system("pause");
   return 0;
}
```

### Recursion (5!)
```
#include <stdio.h>
#include <stdlib.h>
int fac(int);     /* 遞迴函數fac()的原型 */
int main(void)
{
   printf("fac(5)=%d\n", fac(5));   /* 呼叫遞迴函數fac() */

   system("pause");
   return 0;
}

int fac(int n)	/* 自訂函數fac()，計算n! */
{
   if(n>0)
      return (n*fac(n-1));
   else
      return 1;
}
```
