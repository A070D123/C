# 函數設計
### 計算階層
#include <stdio.h>
#include <stdlib.h>

int fac_for(int x)
{
 int total=1;
 for (int i=1;i<=x;i++)
	 {
	  total*=i; 
	 }     
 return total;
}	

int fac_while(int x)
{
 int i=1,total=1;
 while(i<=x)
	  {
	   total*=i++;
	  }	
 return total;
}

int fac_do_while(int x)
{
 int i=1,total=1;
 do
 {
  total*=i++;
 }
 while(i<=x);
 return total; 	
}

int fac_recursion(int x)
{
    if (x==1) return 1;
    return fac_recursion(x-1)*x;
}

int main(void)
{
	int i;
	printf("請輸入數字:");
	scanf("%d",&i);
	
	printf("fac_for       的%d階層=%d\n",i,fac_for(i));
	printf("fac_while     的%d階層=%d\n",i,fac_while(i));
  printf("fac_do_while  的%d階層=%d\n",i,fac_do_while(i));
	printf("fac_recursion 的%d階層=%d\n",i,fac_recursion(i));
	system("pause");
	return 0;
}
